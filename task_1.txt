var a = 1, b = 1, c, d;
c = ++a; alert(c); // 2 - в переменную с мы положили УЖЕ увеличенное на 1 значение а
d = b++; alert(d); // 1 - в переменную d мы положили переменную b, а только потом увеличили b
c = (2+ ++a); alert(c); // 5 ------ 2 + (2+1) / Во второй строке мы увеличили а до двух, и сейчас, мы в с положили 2 + увеличенное на 1 значение с 
d = (2+ b++); alert(d); // 4 - два + b = 4, а потом увеличить b на 1
alert(a); // 3 -- т.к. мы два раза инкрементировали а
alert(b); // 3 -- т.к. мы два раза инкрементировали b
