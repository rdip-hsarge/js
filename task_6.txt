function mathOperations(num1, num2, operation){
 	switch(operation){
      case '+':
        return (num1 + num2);
      case '-':
        return num1 - num2;
      case '*':
        return num1 * num2;
      case '/':
        return num1 / num2;
    }
}

alert(mathOperations(5, 10, '-'));
